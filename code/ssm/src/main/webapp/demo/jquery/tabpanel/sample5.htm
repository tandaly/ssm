在该示例中使用了loader配置，在loader对象直接量里还有一个配置没有加，
就是iframeFlag，如果没有配置该参数或者设置该参数为false，则组件以ajax
方式向后台请求，将请求结果以html方式放在该tab的内容中；若配置iframeFlag
为true，则该tab的内容为iframe，iframe的src是loader的url配置。
后面的例子为iframeFlag是true配置选项的示例
$(document).ready(function(){
        tabpanel= new TabPanel({
            renderTo:'tab', //把tabpanel放在一个div中，div的id为tab
            width:600, //tabpanel宽度
            height:300, //tabpanel高度
            border:'5', //tabpanel的边框，不为'none'即可，若为'none'则没有边框
            active : 1,//默认激活的tab索引数，此处为第二个，索引从0开始
            tabWidth:114,//顶上tab标签页的宽度
            items : [
                {
                  id:'toolbarPlugin0',//标签的id，此id必须在该html中唯一
                  title:'Closable0',//顶上标签的标题
                  loader: {//loader配置
                      url: 'test1.html',//以ajax方式加载标签内容的url
                      param:{abc:'abc'},//参数
                      activateFresh:true//当标签激活时，是否重新刷新页面
                    }
                 },
                {id:'toolbarPlugin1',title:'Closable1',html:'toolbarPluginDiv1'},
                {id:'toolbarPlugin2',title:'Closable2',html:'toolbarPluginDiv2'},
                {id:'toolbarPlugin3',title:'Closable3',html:'toolbarPluginDiv3'},
                {id:'toolbarPlugin4',title:'Closable4',html:'toolbarPluginDiv4'}
            ]
          });
});
