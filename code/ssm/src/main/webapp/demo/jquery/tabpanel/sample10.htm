$(document).ready(function(){
        tabpanel= new TabPanel({
            renderTo:'tab', //把tabpanel放在一个div中，div的id为tab
            width:600, //tabpanel宽度
            height:300, //tabpanel高度
            border:'5', //tabpanel的边框，不为'none'即可，若为'none'则没有边框
            active : 1,//默认激活的tab索引数，此处为第二个，索引从0开始
            tabWidth:114,//顶上tab标签页的宽度
            items : [
                {
                  id:'toolbarPlugin0',//标签的id，此id必须在该html中唯一
                  title:'Closable0',//顶上标签的标题
                  html:"toolbarPlugin0",//标签的内容
                  activate:activate//tab激活后的事件
                 },
                {id:'toolbarPlugin1',title:'Closable1',html:"toolbarPlugin1",activate:activate},
                {id:'toolbarPlugin2',title:'Closable2',html:"toolbarPlugin2",activate:activate},
                {id:'toolbarPlugin3',title:'Closable3',html:"toolbarPlugin3",activate:activate},
                {id:'toolbarPlugin4',title:'Closable4',html:"toolbarPlugin4",activate:activate}
            ]
          });
});
//tab激活后的事件
function activate(tab){
        alert('my title is '+tab.title);
}
