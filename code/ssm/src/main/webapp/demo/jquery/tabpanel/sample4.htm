在同一个items的元素的配置中，html配置，elementId配置和loader配置
是互斥的，也就是说三个里面只能选一个作为配置项，如果同时存在会乱套，
后面的例子为loader配置选项的示例
$(document).ready(function(){
        tabpanel= new TabPanel({
            renderTo:'tab', //把tabpanel放在一个div中，div的id为tab
            width:600, //tabpanel宽度
            height:300, //tabpanel高度
            border:'5', //tabpanel的边框，不为'none'即可，若为'none'则没有边框
            active : 1,//默认激活的tab索引数，此处为第二个，索引从0开始
            tabWidth:114,//顶上tab标签页的宽度
            items : [
                {
                  id:'toolbarPlugin0',//标签的id，此id必须在该html中唯一
                  title:'Closable0',//顶上标签的标题
                  elementId:'toolbarPluginDiv0'//该页面中的一个元素的id，该元素可以是div，span等等
                 },
                {id:'toolbarPlugin1',title:'Closable1',elementId:'toolbarPluginDiv1'},
                {id:'toolbarPlugin2',title:'Closable2',elementId:'toolbarPluginDiv2'},
                {id:'toolbarPlugin3',title:'Closable3',elementId:'toolbarPluginDiv3'},
                {id:'toolbarPlugin4',title:'Closable4',elementId:'toolbarPluginDiv4'}
            ]
          });
});
