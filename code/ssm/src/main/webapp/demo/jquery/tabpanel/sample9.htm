添加tab，如果该tab已经被添加，则将其设置为激活状态，addTab按钮点击后执行的方法：
     function addTab(i){
         //item元素
        var item={
            loadnow:true,//立即显示并激活
             id:'toolbarPlugin',//标签的id，此id必须在该html中唯一
            title:'Closable',//顶上标签的标题
            html:"toolbarPlugin"//标签的内容
        };
        //设置id，title，html等
        item.id=item.id+i;
         item.title=item.title+i;
        item.html=item.html+i;
        //调用组件中的tabpanel的addTab方法添加tab
        tabpanel.addTab(item);
      }
$(document).ready(function(){
        tabpanel= new TabPanel({
            renderTo:'tab', //把tabpanel放在一个div中，div的id为tab
            width:600, //tabpanel宽度
            height:300, //tabpanel高度
            border:'5', //tabpanel的边框，不为'none'即可，若为'none'则没有边框
            active : 1,//默认激活的tab索引数，此处为第二个，索引从0开始
            tabWidth:114,//顶上tab标签页的宽度
            maxLength:10,//可添加的tab元素的最大个数
             items : [
                {
                  id:'toolbarPlugin0',//标签的id，此id必须在该html中唯一
                  title:'Closable0',//顶上标签的标题
                  html:"toolbarPlugin0"//标签的内容
                },
                {id:'toolbarPlugin1',title:'Closable1',html:"toolbarPlugin1"},
                {id:'toolbarPlugin2',title:'Closable2',html:"toolbarPlugin2"},
                {id:'toolbarPlugin3',title:'Closable3',html:"toolbarPlugin3"},
                {id:'toolbarPlugin4',title:'Closable4',html:"toolbarPlugin4"}
            ]
          });
});
